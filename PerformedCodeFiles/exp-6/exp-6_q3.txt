#include <iostream>

using namespace std;

// Base class (A)
class Shape {
public:
    void describe() {
        cout << "I am a shape." << endl;
    }
};

// Derived class 1 (B)
class Circle : public Shape {
public:
    void draw() {
        cout << "Drawing a circle." << endl;
    }
};

// Derived class 2 (C)
class Rectangle : public Shape {
public:
    void draw() {
        cout << "Drawing a rectangle." << endl;
    }
};

int main() {
    cout << "--- Hierarchical Inheritance ---" << endl;

    Circle c1;
    cout << "Circle object:" << endl;
    c1.describe(); // From Shape
    c1.draw();     // From Circle

    Rectangle r1;
    cout << "\nRectangle object:" << endl;
    r1.describe(); // From Shape
    r1.draw();     // From Rectangle

    return 0;
}
