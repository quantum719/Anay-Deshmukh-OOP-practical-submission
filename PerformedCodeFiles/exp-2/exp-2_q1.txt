#include <iostream>
#include <string>

using namespace std;

class city {
public:
    string name;
    long population;

    void accept_data() {
        cout << "Enter city name: ";
        cin.ignore();
        getline(cin, name);
        cout << "Enter population: ";
        cin >> population;
    }
};

int main() {
    const int CITY_COUNT = 5;
    city cities[CITY_COUNT];

    // Accept data for 5 cities
    for (int i = 0; i < CITY_COUNT; ++i) {
        cout << "\nEnter details for city " << (i + 1) << ":" << endl;
        cities[i].accept_data();
    }

    // Find city with highest population
    int max_pop_index = 0;
    for (int i = 1; i < CITY_COUNT; ++i) {
        if (cities[i].population > cities[max_pop_index].population) {
            max_pop_index = i;
        }
    }

    cout << "\n--- City with Highest Population ---" << endl;
    cout << "Name: " << cities[max_pop_index].name << endl;
    cout << "Population: " << cities[max_pop_index].population << endl;

    return 0;
}
